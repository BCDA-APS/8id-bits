"""
Plan that allows for moving to pre-programmed positions seen as strings.

This module provides plans for moving the upstream diagnostics 
among microscope, wavefront sensor, and pind.
"""

import time
from apsbits.core.instrument_init import oregistry

idt_mono = oregistry["idt_mono"]


def select_mono(name: str):

    if name == "111":
        idt_mono.coarse_pitch.move()
        idt_mono.y_slide.move()
        idt_mono.coarse_roll.move()

    if name == "311":
        idt_mono.coarse_pitch.move()
        idt_mono.y_slide.move()
        idt_mono.coarse_roll.move()
    else:
        print("Mono names must be 111 or 311")

    time.sleep(2.0)
